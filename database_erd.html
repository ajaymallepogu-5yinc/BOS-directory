            %% Core Domain
            Departments ||--o{ Employees : "contains"
            AppRoles ||--o{ Employees : "categorizes"
            Employees ||--o{ OrgReportings : "has_reporting"
            
            %% Identity Mapping
            AspNetUsers ||--o| Employees : "links_to"
            AspNetUsers ||--o{ AspNetUserRoles : "has_role"
            AspNetRoles ||--o{ AspNetUserRoles : "linked_to"
            AspNetUsers ||--o{ AspNetUserLogins : "sso_auth"
            AspNetUsers ||--o{ AspNetUserClaims : "has_claims"
            AspNetUsers ||--o{ AspNetUserTokens : "uses_tokens"
            AspNetRoles ||--o{ AspNetRoleClaims : "has_role_claims"

            %% Jira Integration
            JiraProjects ||--o{ JiraIssues : "owns"
            JiraSprints ||--o{ JiraIssues : "sprint_contains"

            Departments {
                int Id PK
                string Name
                string ColorHex
            }

            Employees {
                int Id PK
                string FullName
                string Title
                string Company
                string AvatarUrl
                int ManagerId FK
                int DepartmentId FK
                string AspNetUserId FK
                string APPEmail
                string HRMSEmail
                int APPRoleId FK
            }

            AppRoles {
                int Id PK
                string Name
            }

            OrgReportings {
                int Id PK
                int EmployeeId FK
                int ManagerId FK
                string ReportingType
            }

            DataSourceConfigs {
                int Id PK
                string Mode
                string HrPortalApiUrl
                string HrPortalApiAuthHeaderName
                string HrPortalApiAuthHeaderValue
                string JiraApiUrl
                string JiraUserEmail
                string JiraApiToken
                string IdField
                string FullNameField
                string TitleField
                string CompanyField
                string AvatarUrlField
                string ManagerIdField
                string DepartmentIdField
                string DepartmentNameField
                string DepartmentColorField
            }

            JiraProjects {
                int Id PK
                string Key
                string Name
            }

            JiraSprints {
                int Id PK
                string Name
                string State
                int BoardId
            }

            JiraIssues {
                int Id PK
                string Key
                string Summary
                string Status
                string Assignee
                string Priority
                string Description
                string ExpectedTime
                string ActualTime
                int SprintId FK
                int ProjectId FK
            }
